Відтворення відео контенту в браузері

🎯 Основи Video елемента

Елемент <video> дозволяє вбудовувати відео контент безпосередньо в HTML документ:

/* Базовий синтаксис */ <video controls width="400" height="300"> <source src="../media/video/demo.mp4" type="video/mp4"> <source src="video.webm" type="video/webm"> <source src="video.ogg" type="video/ogg"> Ваш браузер не підтримує HTML5 video. </video> /* З додатковими атрибутами */ <video controls autoplay muted loop poster="thumbnail.jpg" preload="metadata" width="640" height="360"> <source src="../media/video/demo.mp4" type="video/mp4"> <source src="movie.webm" type="video/webm"> <track kind="subtitles" src="subtitles_uk.vtt" srclang="uk" label="Українські субтитри"> <track kind="subtitles" src="subtitles_en.vtt" srclang="en" label="English subtitles"> </video>

⚙️ Атрибути Video елемента

Основні атрибути для управління поведінкою відео:

controls
Показує стандартні елементи управління (play, pause, volume, seek, fullscreen)
<video controls>...</video>
autoplay
Автоматично розпочинає відтворення (потребує muted у більшості браузерів)
<video autoplay muted>...</video>
loop
Зациклює відтворення відео після завершення
<video loop>...</video>
muted
Відключає звук за замовчуванням (необхідно для autoplay)
<video muted>...</video>
poster
Зображення-заставка, що показується до початку відтворення
<video poster="thumb.jpg">...</video>
preload
Контролює попереднє завантаження: "none", "metadata", "auto"
<video preload="metadata">...</video>
width / height
Розміри відео плеєра в пікселях (краще використовувати CSS)
<video width="640" height="360">...</video>
playsinline
Відтворення в лінію на iOS (без переходу в повноекранний режим)
<video playsinline>...</video>
⚠️ Autoplay політика: Сучасні браузери блокують autoplay з звуком. Використовуйте autoplay muted або дочекайтеся взаємодії користувача.

Основні JavaScript методи та властивості

/* Основні методи */ const video = document.getElementById('myVideo'); // Управління відтворенням video.play(); // Розпочати відтворення (повертає Promise) video.pause(); // Призупинити video.load(); // Перезавантажити відео // Властивості відтворення video.currentTime = 30; // Перемотка на 30 секунду video.volume = 0.5; // Гучність 50% video.playbackRate = 1.5; // Швидкість 1.5x video.muted = true; // Вимкнути звук // Інформація про відео console.log(video.duration); // Тривалість в секундах console.log(video.videoWidth); // Ширина відео console.log(video.videoHeight); // Висота відео console.log(video.buffered); // Буферизована частина console.log(video.played); // Відтворена частина // Стан відтворення console.log(video.paused); // Чи на паузі console.log(video.ended); // Чи завершено console.log(video.readyState); // Готовність до відтворення /* Основні події відео */ video.addEventListener('loadstart', () => { console.log('Почато завантаження відео'); }); video.addEventListener('loadedmetadata', () => { console.log('Метадані завантажені'); console.log(`Розмір: ${video.videoWidth}x${video.videoHeight}`); console.log(`Тривалість: ${video.duration} сек`); }); video.addEventListener('loadeddata', () => { console.log('Перший кадр завантажений'); }); video.addEventListener('canplay', () => { console.log('Можна почати відтворення'); }); video.addEventListener('canplaythrough', () => { console.log('Можна відтворювати без зупинок'); }); video.addEventListener('play', () => { console.log('Відтворення розпочато'); }); video.addEventListener('pause', () => { console.log('Відтворення призупинено'); }); video.addEventListener('timeupdate', () => { const progress = (video.currentTime / video.duration) * 100; console.log(`Прогрес: ${progress}%`); }); video.addEventListener('ended', () => { console.log('Відтворення завершено'); }); video.addEventListener('error', (e) => { console.error('Помилка відтворення:', e); });

Fullscreen API

/* Повноекранний режим */ function enterFullscreen(element) { if (element.requestFullscreen) { element.requestFullscreen(); } else if (element.webkitRequestFullscreen) { element.webkitRequestFullscreen(); } else if (element.msRequestFullscreen) { element.msRequestFullscreen(); } } function exitFullscreen() { if (document.exitFullscreen) { document.exitFullscreen(); } else if (document.webkitExitFullscreen) { document.webkitExitFullscreen(); } else if (document.msExitFullscreen) { document.msExitFullscreen(); } } // Перевірка підтримки if (document.fullscreenEnabled) { console.log('Fullscreen підтримується'); } // Події fullscreen document.addEventListener('fullscreenchange', () => { if (document.fullscreenElement) { console.log('Увійшли в fullscreen'); } else { console.log('Вийшли з fullscreen'); } });
