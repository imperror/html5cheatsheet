✅ Form Validation

HTML5 валідація та кастомна перевірка форм

🎯 Типи валідації

Існує кілька способів валідації форм у веб-розробці:

  • HTML5 валідація - вбудована в браузер, працює автоматично
  • JavaScript валідація - кастомна логіка на стороні клієнта
  • Server-side валідація - обов'язкова перевірка на сервері
  • Real-time валідація - перевірка під час введення
  • Progressive enhancement - покращення базової HTML валідації
🔒 Правило безпеки: Ніколи не покладайтеся тільки на клієнтську валідацію! Завжди перевіряйте дані на сервері, оскільки JavaScript можна відключити або обійти.

📋 HTML5 атрибути валідації

HTML5 надає потужні вбудовані можливості валідації:

required
<input required>
Робить поле обов'язковим для заповнення
/* Приклад */ <input type="text" required> <textarea required></textarea> <select required> <option value="">Оберіть...</option> </select>
minlength / maxlength
<input minlength="3" maxlength="20">
Обмежує довжину тексту
/* Приклади */ <input type="text" minlength="2" maxlength="50"> <textarea minlength="10" maxlength="500"></textarea> /* Тільки для text, search, tel, url, password, textarea */
min / max
<input type="number" min="1" max="100">
Обмежує числові значення та дати
/* Приклади */ <input type="number" min="18" max="65"> <input type="range" min="0" max="100"> <input type="date" min="2024-01-01" max="2024-12-31"> <input type="time" min="09:00" max="17:00">
step
<input type="number" step="0.01">
Визначає крок для числових значень
/* Приклади */ <input type="number" step="1"> <!-- Цілі числа --> <input type="number" step="0.01"> <!-- Копійки --> <input type="number" step="5"> <!-- Кратні 5 --> <input type="number" step="any"> <!-- Будь-які -->
pattern
<input pattern="[A-Za-z]{3,}">
Регулярний вираз для перевірки
/* Приклади */ <!-- Тільки літери --> <input pattern="[A-Za-z]+"> <!-- Телефон формату +380XXXXXXXXX --> <input pattern="\+380[0-9]{9}"> <!-- Поштовий індекс України --> <input pattern="[0-9]{5}">
title
<input title="Підказка для користувача">
Кастомне повідомлення про помилку
/* Приклад */ <input type="text" pattern="[A-Za-z]{3,}" title="Мінімум 3 літери, тільки латиниця"> <input type="email" required title="Введіть коректну email адресу">

Кастомна валідація форми

/* Повна кастомна валідація */ function validateForm(form) { const errors = {}; let isValid = true; // Перевіряємо кожне поле const inputs = form.querySelectorAll('input, select, textarea'); inputs.forEach(input => { const fieldErrors = validateField(input); if (fieldErrors.length > 0) { errors[input.name] = fieldErrors; isValid = false; } }); // Відображаємо помилки displayErrors(errors); return isValid; } function validateField(input) { const errors = []; const value = input.value.trim(); // Обов'язкові поля if (input.required && !value) { errors.push('Це поле обов\'язкове'); } // Email валідація if (input.type === 'email' && value) { const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/; if (!emailRegex.test(value)) { errors.push('Некоректний формат email'); } } // Пароль валідація if (input.type === 'password' && value) { if (value.length < 8) { errors.push('Мінімум 8 символів'); } if (!/[0-9]/.test(value)) { errors.push('Має містити цифру'); } if (!/[A-Z]/.test(value)) { errors.push('Має містити велику літеру'); } } return errors; } // Використання form.addEventListener('submit', function(e) { e.preventDefault(); if (validateForm(this)) { // Форма валідна, відправляємо console.log('Форма валідна!'); } else { console.log('Є помилки валідації'); } });

🎨 UX кращі практики

✅ Хороший UX валідації:
  • Inline валідація: Показуйте помилки під час введення
  • Позитивний feedback: Показуйте успіх (зелена галочка)
  • Зрозумілі повідомлення: "Пароль занадто короткий" замість "Invalid"
  • Показуйте вимоги: "Мінімум 8 символів, 1 цифра"
  • Групуйте помилки: Список всіх помилок вгорі форми
  • Фокус на помилці: Автоматично фокусуйте перше неправильне поле
❌ Погані практики:
  • Показувати помилки до того, як користувач почав вводити
  • Неконкретні повідомлення: "Помилка валідації"
  • Валідація тільки при відправці форми
  • Відсутність індикаторів обов'язкових полів
  • Очищення форми при помилці
  • Вимкнення відправки до виправлення всіх помилок

/* Готова до використання функція */ class FormValidator { constructor(form) { this.form = form; this.errors = new Map(); this.init(); } init() { // Real-time валідація this.form.addEventListener('input', (e) => { if (e.target.matches('input, select, textarea')) { this.validateField(e.target); } }); // Валідація при відправці this.form.addEventListener('submit', (e) => { e.preventDefault(); this.validateForm(); }); } validateField(field) { const errors = []; const value = field.value.trim(); // Перевірка обов'язкових полів if (field.hasAttribute('required') && !value) { errors.push(`${this.getFieldLabel(field)} обов'язкове`); } // Перевірка довжини if (value && field.minLength && value.length < field.minLength) { errors.push(`Мінімум ${field.minLength} символів`); } // Спеціальні перевірки по типу if (value) { switch (field.type) { case 'email': if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value)) { errors.push('Некоректний формат email'); } break; case 'tel': if (field.pattern && !new RegExp(field.pattern).test(value)) { errors.push('Некоректний формат телефону'); } break; } } this.updateFieldUI(field, errors); if (errors.length > 0) { this.errors.set(field.name, errors); } else { this.errors.delete(field.name); } return errors.length === 0; } validateForm() { const fields = this.form.querySelectorAll('input, select, textarea'); let isValid = true; fields.forEach(field => { if (!this.validateField(field)) { isValid = false; } }); if (isValid) { this.onSuccess(); } else { this.onError(); } return isValid; } updateFieldUI(field, errors) { const errorContainer = field.parentNode.querySelector('.field-error'); if (errors.length > 0) { field.classList.add('error'); if (errorContainer) { errorContainer.textContent = errors[0]; errorContainer.style.display = 'block'; } } else { field.classList.remove('error'); if (errorContainer) { errorContainer.style.display = 'none'; } } } getFieldLabel(field) { const label = this.form.querySelector(`label[for="${field.id}"]`); return label ? label.textContent.replace('*', '').trim() : 'Поле'; } onSuccess() { console.log('Форма валідна, відправляємо...'); // Тут відправка на сервер } onError() { const firstError = this.form.querySelector('.error'); if (firstError) { firstError.focus(); } } } // Використання const validator = new FormValidator(document.getElementById('myForm'));
